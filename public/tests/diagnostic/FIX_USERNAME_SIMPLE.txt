// ============================================
// FIX USERNAME - VERSION SIMPLE
// ============================================
// Copiez-collez ce code dans la console pour forcer la mise à jour du username

(function() {
  // 1. Récupérer le username du formulaire depuis localStorage
  let usernameFromForm = null;
  try {
    const pendingData = localStorage.getItem('pendingRegisterDataForGoogle');
    if (pendingData) {
      const parsed = JSON.parse(pendingData);
      if (parsed.username && parsed.username !== 'null' && parsed.username !== '' && !parsed.username.includes('@')) {
        usernameFromForm = parsed.username;
        console.log('✅ Username du formulaire trouvé:', usernameFromForm);
      }
    }
  } catch(e) {
    console.error('❌ Erreur:', e);
  }
  
  if (!usernameFromForm) {
    try {
      const pendingData = sessionStorage.getItem('pendingRegisterDataForGoogle');
      if (pendingData) {
        const parsed = JSON.parse(pendingData);
        if (parsed.username && parsed.username !== 'null' && parsed.username !== '' && !parsed.username.includes('@')) {
          usernameFromForm = parsed.username;
          console.log('✅ Username du formulaire trouvé dans sessionStorage:', usernameFromForm);
        }
      }
    } catch(e) {
      console.error('❌ Erreur:', e);
    }
  }
  
  if (!usernameFromForm) {
    console.log('❌ Aucun username du formulaire trouvé');
    return;
  }
  
  // 2. Mettre à jour window.currentUser
  if (window.currentUser) {
    window.currentUser.username = usernameFromForm;
    console.log('✅ window.currentUser.username mis à jour:', window.currentUser.username);
  } else {
    console.log('❌ window.currentUser n\'existe pas');
    return;
  }
  
  // 3. Sauvegarder dans localStorage
  try {
    const currentUserLS = localStorage.getItem('currentUser');
    if (currentUserLS) {
      const parsed = JSON.parse(currentUserLS);
      parsed.username = usernameFromForm;
      localStorage.setItem('currentUser', JSON.stringify(parsed));
      console.log('✅ localStorage.currentUser.username mis à jour:', usernameFromForm);
    }
  } catch(e) {
    console.error('❌ Erreur sauvegarde localStorage:', e);
  }
  
  // 4. Mettre à jour le bloc compte dans le DOM
  const accountName = document.getElementById('account-name');
  if (accountName) {
    accountName.textContent = usernameFromForm;
    console.log('✅ Bloc compte mis à jour dans le DOM:', usernameFromForm);
  } else {
    console.log('❌ account-name non trouvé dans le DOM');
  }
  
  // 5. Appeler updateAccountBlockLegitimately si disponible
  if (typeof window.updateAccountBlockLegitimately === 'function') {
    window.updateAccountBlockLegitimately();
    console.log('✅ updateAccountBlockLegitimately appelée');
  } else {
    console.log('⚠️ updateAccountBlockLegitimately non disponible, mise à jour manuelle effectuée');
  }
  
  console.log('%c✅ Username corrigé !', 'color: #22c55e; font-weight: bold;');
  console.log('   Username utilisé:', usernameFromForm);
  console.log('   window.currentUser.username:', window.currentUser.username);
  console.log('   Bloc compte:', accountName ? accountName.textContent : 'NON TROUVÉ');
})();
